# ============================================================================

# 1. 基础Python依赖：
#    pip install -r requirements.txt

# 2. ROS2依赖：
#    请按照ROS2官方文档安装ROS2 Humble
#    source /opt/ros/humble/setup.bash

# 3. CARLA依赖：
#    下载CARLA并将PythonAPI添加到PYTHONPATH
#    export PYTHONPATH=$PYTHONPATH:/path/to/CARLA/PythonAPI/carla/dist/carla-0.9.13-py3.7-linux-x86_64.egg

# 4. 工作区构建：
#    cd /path/to/your/ros2_workspace
#    colcon build --packages-select multimodal_data_toolkit
